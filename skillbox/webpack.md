# Обнаружила, что половина шрифтов у меня не подключилось
Тут вам нужно добавить лоадер который и подгрузит ваш шрифт. Вот пример конфига:

{
        test: /\.(woff(2)?|eot|ttf|otf|)$/,
        use: [
          {
            loader: "file-loader",
            options: {
              name: `./fonts/${filename("[ext]")}`,
            },
          },
        ],
},
