Тогда вам нужно в функции обработчике клика "добавить дело" добавить:

todoItemForm.form.addEventListener("submit", function (event) {
      event.preventDefault();

      let value = todoItemForm.input.value;

      if (!value) {
        return;

      }
      const data = getData(listName, [])

      let newTodo = createTodo(
        value,
        findMaxValue(data, "id") + 1
      );

      setData(listName, getData(listName, []).concat([newTodo]));


      /* остальной код */
    });


В этом примере используется функция getData:

function getData(key, defaultData) {
   return JSON.parse(localStorage.getItem(key)) || defaultData;
}
и findMaxValue которая сравнивает все значения по ключю и возвращает id последнего элемента:

function findMaxValue(array, key) {
    let max = 0;
    for (let item of array) {
      if (item[key] > max) {
        max = item[key];
      }
    }
    return max;
  }
Тут вы можете так же упростить логику и в findMaxValue возвращать array.length.
